-- Enable UUID extension
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Purchases Table
CREATE TABLE purchases (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    date DATE NOT NULL,
    supplier TEXT NOT NULL,
    paddy_type TEXT NOT NULL,
    quantity_kg NUMERIC(10, 2) NOT NULL,
    total_amount NUMERIC(12, 2) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Production Table
CREATE TABLE production (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    date DATE NOT NULL,
    paddy_type TEXT NOT NULL,
    input_paddy NUMERIC(10, 2) NOT NULL,
    rice_output NUMERIC(10, 2) NOT NULL,
    yield_percentage NUMERIC(5, 2) GENERATED ALWAYS AS ((rice_output / NULLIF(input_paddy, 0)) * 100) STORED,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Sales Table
CREATE TABLE sales (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    date DATE NOT NULL,
    customer TEXT NOT NULL,
    phone TEXT,
    rice_type TEXT NOT NULL,
    quantity NUMERIC(10, 2) NOT NULL,
    amount NUMERIC(12, 2) NOT NULL,
    payment_type TEXT NOT NULL CHECK (payment_type IN ('Cash', 'Loan', 'BankTransfer')),
    loan_status TEXT CHECK (loan_status IN ('Paid', 'Unpaid', NULL)),
    bank_name TEXT,
    bank_account TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Expenses Table
CREATE TABLE expenses (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    date DATE NOT NULL,
    category TEXT NOT NULL,
    description TEXT,
    amount NUMERIC(12, 2) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Create Indexes
CREATE INDEX idx_purchases_date ON purchases(date);
CREATE INDEX idx_production_date ON production(date);
CREATE INDEX idx_sales_date ON sales(date);
CREATE INDEX idx_sales_payment_type ON sales(payment_type);
CREATE INDEX idx_expenses_date ON expenses(date);

-- Enable Row Level Security
ALTER TABLE purchases ENABLE ROW LEVEL SECURITY;
ALTER TABLE production ENABLE ROW LEVEL SECURITY;
ALTER TABLE sales ENABLE ROW LEVEL SECURITY;
ALTER TABLE expenses ENABLE ROW LEVEL SECURITY;

-- RLS Policies (Allow all operations for now - you can customize later)
CREATE POLICY "Enable all operations for purchases" ON purchases FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "Enable all operations for production" ON production FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "Enable all operations for sales" ON sales FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "Enable all operations for expenses" ON expenses FOR ALL USING (true) WITH CHECK (true);

-- Update timestamp function
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

-- Create triggers for updated_at
CREATE TRIGGER update_purchases_updated_at BEFORE UPDATE ON purchases
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_production_updated_at BEFORE UPDATE ON production
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_sales_updated_at BEFORE UPDATE ON sales
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_expenses_updated_at BEFORE UPDATE ON expenses
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
